1. 工厂模式


$(function(){
//工厂模式
 function createPerson(name,age,job){
  o=new Object();      //此处为Object（）
  o.name=name;
  o.age=age;
  o.job=job;
  o.sayName=function(){
   alert(this.name);
  }
  return o;
 }
 var Johnson=createPerson("Johnson",21,"student");
 var Penny=createPerson("Penny",25,"actress");
 Johnson.sayName();
 Penny.sayName();

}
)
