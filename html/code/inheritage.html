创建对象

1. 原型链继承：将一个对象的原型函数设置为另一个对象的实例来实现继承;
使用该方式时SubType.prototype的constructor是指向SperType的，可以这么理解，当调用SubType.prorotype.constructor时，系统先搜索作为原型函数的SuperType的实例，该实例中不存在这个属性，因此继续搜索SuperType的原型，此时可以搜索到，但是却是指向SuperType的，因为只有新定义函数时才会生成一个对应的prototype对象同时默认有一个constructor属性,而在原型链继承中并没有新定义函数，因此不会生成新的constructor属性
$(function(){
 function Supertype(){
  this.property=true;
 }
 Supertype.prototype.getSuperValue=function(){
  return this.property;
 }
 function Subtype(){
  this.subproperty=false;
 }
 
 Subtype.prototype=new Supertype();
 Subtype.prototype.getSubvalue=function(){
  return this.subproperty;
 };
 
 var instance=new Subtype;
 alert(instance.getSuperValue());
}
) 

2. 构造函数方式继承：
$(function(){
 
//借用构造函数模式，无法实现代码复用
 function SuperType(name)
 {
  this.name=name;
 }

 function SubType(name,age,job){
  SuperType.call(this,name);
  this.age=age;
  this.job=job;
 }
 var instance=new SubType("John",21,"doctor");
 alert(instance.name);
}
) 

3.组合继承（原型链+构造函数）
$(function(){
 
//组合继承
$(function(){
 
//组合继承
 function SuperType(name)
 {
  this.name=name;
  this.colors=["red","blue","green"];
 }
 
 SuperType.prototype.sayName=function(){
  alert(this.name);
 };
 
 function SubType(name,age,job){
  SuperType.call(this,name);
  this.age=age;
  this.job=job;
 }
 SubType.prototype=new SuperType();
 alert(SubType.prototype.constructor);  //此时该constructor指针指向SuperType
 SubType.prototype.constructor=SubType;
 alert(SubType.prototype.constructor);   //此时该constructor指针指向SubType
 SubType.prototype.sayAge=function(){
  alert(this.age);
 }
 
 var instance1=new SubType("John",21,"doctor");
 instance1.colors.push("black");
 alert(instance1.colors);
 instance1.sayName();
 instance1.sayAge();
 
 var instance2=new SubType("Holmes",23,"detective");
 alert(instance2.colors);
 instance2.sayName();
 instance2.sayAge();

 
}
